        def determine_superclass(opts)
          if type = opts[:type]
            self[type]
          elsif opts[:location] =~ /spec(\\|\/)(#{@example_group_types.keys.sort_by{|k| k.to_s.length}.reverse.join('|')})/
            self[$2 == '' ? nil : $2.to_sym]
          else
            self[nil]
          end
        end
